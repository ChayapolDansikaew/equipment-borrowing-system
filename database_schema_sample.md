# ตัวอย่างโครงสร้างฐานข้อมูล (Database Schema)

เอกสารนี้แสดงซอร์สโค้ด SQL ที่ใช้สำหรับสร้างตารางข้อมูลในระบบหลังบ้านบน **Supabase (PostgreSQL)** ซึ่งประกอบไปด้วย 4 ตารางหลัก ตามที่มีการระบุในเอกสารการออกแบบ ได้แก่ `users`, `equipments`, `transactions`, และ `penalties`

## 1. ตาราง users
ใช้สำหรับเก็บข้อมูลผู้ใช้งานระบบ สถานะสิทธิ์ (Role) และยอดสะสมบทลงโทษ (Strikes) รวมถึงสถานะการแบน

```sql
CREATE TABLE users (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    username TEXT NOT NULL UNIQUE,
    password TEXT NOT NULL,
    role TEXT NOT NULL CHECK (role IN ('admin', 'user')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- ส่วนของระบบบทลงโทษ (Penalty System)
    total_strikes INTEGER DEFAULT 0,
    is_banned BOOLEAN DEFAULT FALSE,
    ban_until TIMESTAMP WITH TIME ZONE,
    ban_reason TEXT
);
```

## 2. ตาราง equipments
ใช้สำหรับเก็บข้อมูลอุปกรณ์ หมวดหมู่ และสถานะความพร้อมในการถูกยืม

```sql
CREATE TABLE equipments (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    type TEXT NOT NULL,
    image_url TEXT NOT NULL,
    status TEXT NOT NULL CHECK (status IN ('available', 'borrowed')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

## 3. ตาราง transactions
ใช้สำหรับเก็บข้อมูลรายการยืม-คืนอุปกรณ์ และวันเวลาเริ่มต้น-สิ้นสุดของการยืม

```sql
CREATE TABLE transactions (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    equipment_id BIGINT REFERENCES equipments(id),
    borrower_name TEXT NOT NULL,
    borrow_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    return_date TIMESTAMP WITH TIME ZONE,
    start_date TIMESTAMP WITH TIME ZONE, -- สำหรับเช็ควันที่เริ่มยืม
    end_date TIMESTAMP WITH TIME ZONE,   -- สำหรับเช็ควันที่สิ้นสุด (กำหนดคืน)
    status TEXT NOT NULL CHECK (status IN ('active', 'returned'))
);
```

## 4. ตาราง penalties
ใช้สำหรับเก็บข้อมูลประวัติบทลงโทษ ระดับความรุนแรง และค่าชดใช้ ในกรณีที่ผู้ใช้ทำผิดกฎ (เช่น คืนช้า, สูญหาย, เสียหาย)

```sql
CREATE TABLE penalties (
    id SERIAL PRIMARY KEY,
    user_id VARCHAR(255) NOT NULL,           -- อ้างอิง username ผู้กระทำผิด
    transaction_id INTEGER REFERENCES transactions(id),
    equipment_id INTEGER REFERENCES equipments(id),
    
    -- ประเภทความผิดและระดับ
    penalty_type VARCHAR(50) NOT NULL,       -- เช่น 'late_return', 'minor_damage', 'lost'
    severity VARCHAR(20) NOT NULL,           -- เช่น 'low', 'medium', 'high', 'critical'
    days_late INTEGER DEFAULT 0,             -- จำนวนวันที่ล่าช้า
    
    -- บทลงโทษและค่าชดใช้
    description TEXT,                        
    strikes_given INTEGER DEFAULT 0,          -- จำนวน strike ที่บันทึกเพิ่มให้ user
    compensation_amount DECIMAL(10,2) DEFAULT 0, -- ค่าชดใช้ (ถ้ามี)
    compensation_status VARCHAR(20) DEFAULT 'none', 
    
    created_by VARCHAR(255),                  -- แอดมินที่ทำรายการ
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    notes TEXT                                
);
```
